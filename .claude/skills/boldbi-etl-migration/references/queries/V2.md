WITH user_acl AS
  (SELECT ud.organization,
          ud.uuid
   FROM user_details ud
   WHERE organization = @{{:OrganizationParameter}}
     AND is_active = 'true'
     AND job_location NOT IN ('KNOW',
                              'All')
     AND job_location NOT ILIKE 'Head Office%'
     AND (
            (SELECT is_super_admin
             FROM user_details
             WHERE uuid = @{{:UuidParameter}})
          OR uuid IN
            (SELECT DISTINCT user_id
             FROM user_groups ug1
             WHERE ug1.group_id IN
                 (SELECT group_id
                  FROM user_groups ug2
                  WHERE ug2.user_id =@{{:UuidParameter}}
                    AND ug2.has_access = TRUE)
               AND ug1.is_active = TRUE))
   ),
     td AS
  (SELECT id AS organization,
          tzoffset, interval '1 min'*tzoffset AS diff
   FROM organizations
   WHERE id = @{{:OrganizationParameter}}
   limit 1)
SELECT la.organization,
       la.course_name,
       la.uuid,
       la.emp_name,
       la.emp_id,
       la.division,
       la.sub_division,
       la.LOCATION,
       la.department,
       la.designation,
       la.job_type,
       la.sent_at + td.diff AS shared_at,
       la.completed_at + td.diff AS completed_at,
       la.course_status,
       la.quiz_status,
       la.score_in_pct,
       la.course_knid,
       la.user_knid
  from learn_analytics la, td
  WHERE sent_at BETWEEN @{{:Date Range.START}}::TIMESTAMP AND @{{:Date Range.END}}::TIMESTAMP + interval '1 day'
  and la.uuid in (select uuid from user_acl)
